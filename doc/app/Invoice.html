def create(url_prefix, filename)
  object = new(url_prefix, filename)

  begin
    object.invoice_id = KuluService::API.new.create_invoice(object.storage_key)
  rescue HTTPService::Error => e
    object.errors.add(:base, e.message)
  end

  object
end
